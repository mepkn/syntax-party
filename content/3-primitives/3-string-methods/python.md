```python
s = "  Hello, World!  "

len(s)                      # 17
s.strip()                   # "Hello, World!"
s.upper()                   # "  HELLO, WORLD!  "
s.lower()                   # "  hello, world!  "
s[0]                        # " "
s[2:7]                      # "Hello"
"World" in s                # True
s.replace("World", "Py")    # "  Hello, Py!  "
s.split(", ")               # ["  Hello", "World!  "]
```
